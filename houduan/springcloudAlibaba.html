                        # 3.4读取多个配置文件

                        !!!!!!!!!!!!!!!!!!!!!!!!侧重与读不同组别的文件!!!!!!!!!!!!!!!!!

                        同时读同一个命名空间的不同组

                        server:
  port: 8082
spring:
  application:
    name: nacos-config-test
  cloud:
    nacos:
      config:
        server-addr: localhost:8848
        username: nacos
        password: nacos
        namespace: eed441f5-6a24-4ec1-b489-03c577c1f2a9 # 读多配置 只能在同一个命名空间下
        file-extension: yml
        extension-configs:  # 可以读多个配置文件 同一命名空间 可以不同组
          - dataId: user-center-dev.yml
            group: A_GROUP
            refresh: true  # 开启动态刷新
          - dataId: member-center-dev.yml
            group: B_GROUP
            refresh: true

                        bootstrap.yml
                        应用名称：spring.application.name

                        spring:
  application:
    name: nacos-config-test
  cloud:
    nacos:
      config:
        server-addr: localhost:8848
        username: nacos
        password: nacos
        namespace: eed441f5-6a24-4ec1-b489-03c577c1f2a9 # 读多配置 只能在同一个命名空间下
        file-extension: yml
        prefix: user-center
        group: A_GROUP
        shared-configs: # 共享配置文件
          - dataId: application-dev.yml
            group: C_GROUP

        extension-configs:  # 可以读多个配置文件 同一命名空间 可以不同组
          - dataId: user-center-dev.yml
            group: A_GROUP
            refresh: true  # 开启动态刷新
          - dataId: member-center-dev.yml
            group: B_GROUP
            refresh: true
  profiles:
    active: dev
